💻 Step 2：第一个 Chain——翻译 + 摘要

Python 
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from dotenv import load_dotenv

load_dotenv()

# 1. 初始化模型（也可以用通义千问：ChatTongyi）
llm = ChatOpenAI(
    model="gpt-4o-mini",  # 便宜又好用
    temperature=0.3           # 低温度让输出更稳定
)

# 2. 定义提示词模板
prompt = ChatPromptTemplate.from_messages([
    ("system", "你是一个专业的翻译和摘要助手。"),
    ("user", "请将以下文本翻译成英文，然后用一句话总结主要内容：\n{text}")
])

# 3. 构建 Chain：提示词 → LLM → 输出解析
chain = prompt | llm | StrOutputParser()

# 4. 运行 Chain
result = chain.invoke({
    "text": "人工智能正在深刻改变我们的生活方式。从智能手机中的语音助手，到自动驾驶汽车，再到医疗诊断系统，AI技术已经渗透到日常生活的方方面面。"
})
print(result)

💻 Step 3：进阶——使用 SequentialChain 串联多个步骤

Python 
from langchain_core.runnables import RunnableLambda

# 拆分为两个独立的 Chain，再用管道连接
translate_prompt = ChatPromptTemplate.from_template(
    "将以下中文翻译成英文：{text}"
)
summarize_prompt = ChatPromptTemplate.from_template(
    "用一句话总结以下英文内容：{english_text}"
)

# 定义两个 Chain
translate_chain = translate_prompt | llm | StrOutputParser()
summarize_chain = summarize_prompt | llm | StrOutputParser()

# 用一个函数把两个 Chain 串起来
def translate_then_summarize(inputs):
    english = translate_chain.invoke(inputs)
    summary = summarize_chain.invoke({"english_text": english})
    return {"english": english, "summary": summary}

full_chain = RunnableLambda(translate_then_summarize)
result = full_chain.invoke({
    "text": "量子计算利用量子比特的叠加态和纠缠特性，能够在某些特定问题上实现远超经典计算机的计算能力。"
})
print(f"英文：{result['english']}\n摘要：{result['summary']}")
💡 关键要点：LangChain 的核心思想是"管道模式"——每个组件像一个管道段，数据从左到右流经每个段。`prompt | llm | parser` 这种语法称为 LCEL（LangChain Expression Language），是 LangChain 最推荐的使用方式。

💻 Step 3：加载模型并配置 LoRA

Python 
from transformers import AutoModelForCausalLM, AutoTokenizer, TrainingArguments, Trainer
from peft import LoraConfig, get_peft_model, TaskType
from datasets import load_dataset

# 1. 加载模型和分词器
model_name = "Qwen/Qwen2.5-0.5B-Instruct"
tokenizer = AutoTokenizer.from_pretrained(model_name, trust_remote_code=True)
model = AutoModelForCausalLM.from_pretrained(
    model_name,
    device_map="auto",        # 自动分配 GPU/CPU
    trust_remote_code=True
)

# 2. 配置 LoRA
lora_config = LoraConfig(
    task_type=TaskType.CAUSAL_LM,
    r=8,              # LoRA 秩：越大模型容量越大，但训练越慢
    lora_alpha=32,     # 缩放因子
    lora_dropout=0.1,  # Dropout 防过拟合
    target_modules=["q_proj", "k_proj", "v_proj", "o_proj"],  # 对注意力层加 LoRA
)

# 3. 应用 LoRA 到模型
model = get_peft_model(model, lora_config)
model.print_trainable_parameters()  # 通常只训练 0.1%~1% 的参数！

💻 Step 4：训练并保存

Python 
# 加载数据集
dataset = load_dataset("json", data_files="train.jsonl", split="train")

# 格式化函数：将数据转换为模型需要的格式
def format_example(example):
    text = f"### 指令：{example['instruction']}\n### 输入：{example['input']}\n### 输出：{example['output']}"
    return tokenizer(text, truncation=True, max_length=512)

dataset = dataset.map(format_example)

# 训练配置
training_args = TrainingArguments(
    output_dir="./qwen-lora-output",
    num_train_epochs=3,
    per_device_train_batch_size=2,
    gradient_accumulation_steps=4,
    learning_rate=2e-4,
    logging_steps=10,
    save_steps=50,
    fp16=True,  # 混合精度训练，省显存
)

trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=dataset,
)
trainer.train()

# 保存 LoRA 权重（不是完整模型，很小！）
model.save_pretrained("./qwen-lora-adapter")
tokenizer.save_pretrained("./qwen-lora-adapter")
print("✅ 训练完成！LoRA 权重已保存")
💡 关键要点：LoRA 的核心思想是"冻结原始权重，只训练一个小矩阵"。原始 0.5B 模型有 5 亿参数，而 LoRA 只训练约 50 万参数（0.1%），效果却接近全量微调。训练好的 LoRA 权重只是一个几 MB 的小文件，可以像插件一样加载和卸载。

💻 Step 2：创建向量数据库并插入文档

Python 
import chromadb
from sentence_transformers import SentenceTransformer

# 1. 初始化 ChromaDB 客户端
client = chromadb.PersistentClient(path="./my_chroma_db")

# 2. 加载 Embedding 模型（将文本转为向量）
# 这个模型会将任意文本映射为 384 维的向量
embedder = SentenceTransformer("all-MiniLM-L6-v2")

# 3. 创建集合（类似关系数据库的"表"）
collection = client.get_or_create_collection(
    name="my_documents",
    metadata={"description": "AI 知识库文档"}
)

# 4. 准备文档并插入
documents = [
    "大语言模型（LLM）是基于 Transformer 架构的超大规模神经网络，通过海量文本训练获得语言理解和生成能力。",
    "知识图谱是一种以图结构组织知识的技术，节点代表实体，边代表关系，支持逻辑推理和语义查询。",
    "RAG（检索增强生成）结合了信息检索和文本生成，先检索相关文档再让 LLM 基于文档生成回答，大幅减少幻觉。",
    "Python 是一种高级编程语言，以其简洁的语法和丰富的库生态而受到广泛欢迎。",
    "LoRA 是一种参数高效的微调方法，通过在预训练模型旁添加低秩矩阵来实现任务适配，大幅降低训练成本。",
]

# 转为向量并插入
embeddings = embedder.encode(documents).tolist()
collection.add(
    documents=documents,
    embeddings=embeddings,
    ids=[f"doc_{i}" for i in range(len(documents))]
)

print(f"✅ 已插入 {collection.count()} 条文档")

💻 Step 3：语义搜索

Python 
# 输入查询
query = "如何减少大模型的幻觉问题？"

# 将查询也转为向量，然后搜索最相似的文档
results = collection.query(
    query_embeddings=embedder.encode([query]).tolist(),
    n_results=3  # 返回最相似的 3 条
)

print(f"🔍 查询：{query}\n")
for i, (doc, distance) in enumerate(zip(results["documents"][0], results["distances"][0])):
    print(f"#{i+1} [相似度: {1-distance:.3f}] {doc[:80]}...")

# 输出示例：
# #1 [相似度: 0.82] RAG（检索增强生成）结合了信息检索和文本生成...
# #2 [相似度: 0.71] 大语言模型（LLM）是基于 Transformer 架构的超大规模...
# #3 [相似度: 0.35] LoRA 是一种参数高效的微调方法...
💡 关键要点：向量数据库的核心是"语义相似"而非"关键词匹配"。即使查询中没有出现"RAG"这个词，系统也能找到关于 RAG 的文档——因为"减少幻觉"和"RAG"在语义上高度相关。这就是向量搜索相比传统全文搜索的革命性优势。

💻 Step 1：定义工具函数

Python 
import json
import math
from datetime import datetime

# 定义工具函数（实际项目中可能调用外部 API）
def get_weather(city: str) -> str:
    """获取指定城市的天气（模拟）"""
    weather_data = {
        "北京": "晴，25°C，湿度 40%",
        "上海": "多云，28°C，湿度 65%",
        "深圳": "阵雨，30°C，湿度 80%",
    }
    return weather_data.get(city, f"未找到{city}的天气数据")

def calculate(expression: str) -> str:
    """执行数学计算"""
    try:
        result = eval(expression, {"__builtins__": {}}, math.__dict__)
        return f"计算结果：{expression} = {result}"
    except Exception as e:
        return f"计算错误：{str(e)}"

def get_current_time() -> str:

from openai import OpenAI

client = OpenAI()  # 使用 OPENAI_API_KEY 环境变量

def run_agent(user_query: str) -> str:
    messages = [{"role": "user", "content": user_query}]

    # 第一轮：让 LLM 决定是否需要调用工具
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=messages,
        tools=TOOL_SCHEMAS,
        tool_choice="auto"  # 自动决定是否调用工具
    )

    msg = response.choices[0].message

    # 如果 LLM 决定调用工具
    if msg.tool_calls:
        for tool_call in msg.tool_calls:
            func_name = tool_call.function.name
            func_args = json.loads(tool_call.function.arguments)

            print(f"🔧 调用工具：{func_name}({func_args})")

            # 执行工具函数
            func = TOOLS[func_name]
            result = func(**func_args)

            # 将工具执行结果返回给 LLM
            messages.append(msg)
            messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": result
            })

        # 第二轮：让 LLM 基于工具结果生成最终回答
        final_response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=messages
        )
        return final_response.choices[0].message.content

    return msg.content

# 测试

💡 关键要点：Function Calling 的本质是让 LLM 输出结构化的 JSON（而非自然语言），然后由程序解析 JSON 并实际执行函数。LLM 本身不执行任何代码——它只是"选择"调用哪个工具和传什么参数。执行结果再返回给 LLM，让它整合成自然语言回答。这就是 Agent 决策循环的基础。

ReAct = Reasoning + Acting。Agent 在每一轮中：思考 → 决定是否调用工具 → 调用 → 观察结果 → 继续思考。

# agent.py —— ReAct Agent 核心循环
import json
import re
from openai import OpenAI  # 或任何兼容 OpenAI API 的服务

client = OpenAI(base_url="http://localhost:11434/v1", api_key="ollama")

SYSTEM_PROMPT = """你是一个智能助手，可以使用工具来完成任务。

## 可用工具
{tools_description}

## 响应格式（严格遵守）
如果你想调用工具，请用以下格式：

{"name": "工具名称", "arguments": {"参数名": "参数值"}}


如果你想直接回答用户，请用以下格式：

你的回答内容


注意：每次只能调用一个工具。收到工具结果后，可以继续调用工具或给出最终回答。"""

def format_tools_for_prompt(tools):
    """将工具列表转为 LLM 可理解的描述"""
    desc = []
    for t in tools:
        desc.append(f"- {t.name}: {t.description}")
        desc.append(f"  参数: {json.dumps(t.parameters, ensure_ascii=False)}")
    return "\n".join(desc)

def extract_tool_call(text: str):
    """从 LLM 输出中提取工具调用"""
    match = re.search(r'\s*(\{.*?\})\s*', text, re.DOTALL)
    if match:
        return json.loads(match.group(1))
    return None

def extract_answer(text: str):
    """从 LLM 输出中提取最终回答"""
    match = re.search(r'\s*(.*?)\s*', text, re.DOTALL)
    if match:
        return match.group(1).strip()
    return text.strip()

def run_agent(user_query: str, max_rounds: int = 10):
    """运行 Agent 主循环"""
    messages = [
        {"role": "system", "content": SYSTEM_PROMPT.format(
            tools_description=format_tools_for_prompt(tools)
        )},
        {"role": "user", "content": user_query}
    ]
    
    tool_map = {t.name: t for t in tools}
    
    for round_num in range(max_rounds):
        print(f"\n{'='*50}")
        print(f"🔄 第 {round_num + 1} 轮")
        
        response = client.chat.completions.create(
            model="qwen2.5:7b",
            messages=messages,
            temperature=0.1
        )
        reply = response.choices[0].message.content
        print(f"💭 LLM 输出:\n{reply}")
        
        # 检查是否要调用工具
        tool_call = extract_tool_call(reply)
        if tool_call:
            tool_name = tool_call["name"]
            tool_args = tool_call.get("arguments", {})
            
            if tool_name not in tool_map:
                result = f"错误：未知工具 '{tool_name}'"
            else:
                print(f"🔧 调用工具: {tool_name}({tool_args})")
                result = tool_map[tool_name].run(**tool_args)
                print(f"📊 工具返回: {result}")
            
            # 将工具结果加入对话历史
            messages.append({"role": "assistant", "content": reply})
            messages.append({"role": "user", "content": f"{result}"})
            continue
        
        # 检查是否给出最终回答
        answer = extract_answer(reply)
        if answer:
            print(f"✅ 最终回答: {answer}")
            return answer
        
        # 如果既没有工具调用也没有回答，把 LLM 输出作为普通消息继续
        messages.append({"role": "assistant", "content": reply})
        messages.append({"role": "user", "content": "请继续。如果需要调用工具，使用  格式；如果已有足够信息，使用  格式给出最终回答。"})
    
    return "达到最大轮次限制，Agent 未能完成任务。"

Step 2：实现 Transformer 核心组件

# model.py —— 从零实现 GPT 模型
import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class Head(nn.Module):
    """单个注意力头"""
    def __init__(self, n_embd, head_size, block_size, dropout=0.1):
        super().__init__()
        self.key = nn.Linear(n_embd, head_size, bias=False)
        self.query = nn.Linear(n_embd, head_size, bias=False)
        self.value = nn.Linear(n_embd, head_size, bias=False)
        self.register_buffer('tril', torch.tril(torch.ones(block_size, block_size)))
        self.dropout = nn.Dropout(dropout)

    def forward(self, x):
        B, T, C = x.shape
        k = self.key(x)    # (B, T, head_size)
        q = self.query(x)  # (B, T, head_size)
        v = self.value(x)  # (B, T, head_size)
        
        # 计算注意力分数：Q @ K^T / sqrt(d_k)
        wei = q @ k.transpose(-2, -1) * (k.shape[-1] ** -0.5)
        # 因果掩码：防止看到未来 token
        wei = wei.masked_fill(self.tril[:T, :T] == 0, float('-inf'))
        wei = F.softmax(wei, dim=-1)
        wei = self.dropout(wei)
        return wei @ v

class MultiHeadAttention(nn.Module):
    """多头注意力：并行运行多个注意力头"""
    def __init__(self, n_embd, n_head, head_size, block_size, dropout=0.1):
        super().__init__()
        self.heads = nn.ModuleList([
            Head(n_embd, head_size, block_size, dropout) 
            for _ in range(n_head)
        ])
        self.proj = nn.Linear(n_head * head_size, n_embd)
        self.dropout = nn.Dropout(dropout)

    def forward(self, x):
        out = torch.cat([h(x) for h in self.heads], dim=-1)
        return self.dropout(self.proj(out))

class FeedForward(nn.Module):
    """前馈网络：两层 MLP + GELU 激活"""
    def __init__(self, n_embd, dropout=0.1):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(n_embd, 4 * n_embd),
            nn.GELU(),
            nn.Linear(4 * n_embd, n_embd),
            nn.Dropout(dropout),
        )

    def forward(self, x):
        return self.net(x)

class Block(nn.Module):
    """一个 Transformer Block = Attention + FeedForward"""
    def __init__(self, n_embd, n_head, block_size, dropout=0.1):
        super().__init__()
        head_size = n_embd // n_head
        self.ln1 = nn.LayerNorm(n_embd)
        self.attn = MultiHeadAttention(n_embd, n_head, head_size, block_size, dropout)
        self.ln2 = nn.LayerNorm(n_embd)
        self.ffwd = FeedForward(n_embd, dropout)

    def forward(self, x):
        x = x + self.attn(self.ln1(x))  # 残差连接 + 注意力
        x = x + self.ffwd(self.ln2(x))  # 残差连接 + 前馈
        return x

Step 3：组装完整模型 + 训练

# 完整的 GPT 模型
class GPT(nn.Module):
    def __init__(self, vocab_size, n_embd=256, n_head=4, n_layer=6, 
                 block_size=128, dropout=0.1):
        super().__init__()
        self.token_embedding = nn.Embedding(vocab_size, n_embd)
        self.position_embedding = nn.Embedding(block_size, n_embd)
        self.blocks = nn.Sequential(*[
            Block(n_embd, n_head, block_size, dropout) 
            for _ in range(n_layer)
        ])
        self.ln_f = nn.LayerNorm(n_embd)
        self.lm_head = nn.Linear(n_embd, vocab_size)
        self.block_size = block_size

    def forward(self, idx):
        B, T = idx.shape
        tok_emb = self.token_embedding(idx)  # (B, T, n_embd)
        pos = torch.arange(T, device=idx.device)
        pos_emb = self.position_embedding(pos)  # (T, n_embd)
        x = tok_emb + pos_emb
        x = self.blocks(x)
        x = self.ln_f(x)
        return self.lm_head(x)  # (B, T, vocab_size)

    def generate(self, idx, max_new_tokens, temperature=0.8):
        """生成文本（自回归）"""
        for _ in range(max_new_tokens):
            # 截断到 block_size
            idx_cond = idx[:, -self.block_size:]
            logits = self(idx_cond)  # (B, T, vocab_size)
            logits = logits[:, -1, :] / temperature  # 只取最后一个位置
            probs = F.softmax(logits, dim=-1)
            idx_next = torch.multinomial(probs, num_samples=1)  # 采样
            idx = torch.cat((idx, idx_next), dim=1)
        return idx

# ===== 训练循环 =====
model = GPT(vocab_size, n_embd=256, n_head=4, n_layer=6)
optimizer = torch.optim.AdamW(model.parameters(), lr=3e-4)

for step in range(5000):
    xb, yb = get_batch('train', block_size=128, batch_size=32)
    logits = model(xb)
    loss = F.cross_entropy(
        logits.view(-1, vocab_size), 
        yb.view(-1)
    )
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()
    
    if step % 500 == 0:
        print(f"Step {step}: loss = {loss.item():.4f}")

# 生成文本
context = torch.tensor([encode("ROMEO:")], dtype=torch.long)
generated = model.generate(context, max_new_tokens=200, temperature=0.8)
print(decode(generated[0].tolist()))
🔑 关键要点：
1. 这个 ~200 行的模型就是 GPT 的核心，真正的 GPT-3 只是放大了参数（更多层、更宽维度）
2. 因果掩码（causal mask）是 GPT 单向生成的关键——每个 token 只能看到前面的 token
3. 残差连接让深层网络可以训练，没有它，Transformer 堆叠不了很多层
4. 训练 5000 步后，模型已经能生成类似莎士比亚风格的文本（虽然还不够好）
5. 在 CPU 上也能跑（约 30 分钟），有 GPU 会快很多

Step 1：文档分块器

# chunker.py —— 智能文档分块
import re
from typing import List

class TextChunker:
    """将长文档切分为适合检索的小块"""
    
    def __init__(self, chunk_size: int = 500, chunk_overlap: int = 100):
        """
        chunk_size: 每块最大字符数
        chunk_overlap: 相邻块之间的重叠字符数（防止信息断裂）
        """
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
    
    def split_by_paragraph(self, text: str) -> List[str]:
        """按段落分块（优先保持语义完整性）"""
        paragraphs = re.split(r'\n\s*\n', text)
        chunks = []
        current_chunk = ""
        
        for para in paragraphs:
            para = para.strip()
            if not para:
                continue
            
            # 如果加入当前段落后超过 chunk_size，先保存当前块
            if len(current_chunk) + len(para) > self.chunk_size and current_chunk:
                chunks.append(current_chunk.strip())
                # 保留 overlap 部分
                overlap_text = current_chunk[-self.chunk_overlap:] if self.chunk_overlap > 0 else ""
                current_chunk = overlap_text + "\n\n" + para
            else:
                current_chunk += ("\n\n" + para) if current_chunk else para
        
        if current_chunk.strip():
            chunks.append(current_chunk.strip())
        
        return chunks
    
    def split_by_sentence(self, text: str) -> List[str]:
        """按句子分块（更细粒度，适合问答场景）"""
        sentences = re.split(r'(?<=[。！？.!?])\s*', text)
        chunks = []
        current_chunk = ""
        
        for sent in sentences:
            sent = sent.strip()
            if not sent:
                continue
            if len(current_chunk) + len(sent) > self.chunk_size and current_chunk:
                chunks.append(current_chunk.strip())
                current_chunk = sent
            else:
                current_chunk += sent
        
        if current_chunk.strip():
            chunks.append(current_chunk.strip())
        
        return chunks

# 使用示例
chunker = TextChunker(chunk_size=500, chunk_overlap=100)
with open('knowledge_base.txt', 'r', encoding='utf-8') as f:
    doc = f.read()

chunks = chunker.split_by_paragraph(doc)
print(f"文档被切分为 {len(chunks)} 个块")
for i, chunk in enumerate(chunks[:3]):
    print(f"\n--- Chunk {i+1} ({len(chunk)} 字符) ---")
    print(chunk[:200] + "...")

Step 2：向量化 + 检索

# retriever.py —— 向量化存储与语义检索
import numpy as np
from openai import OpenAI

class SimpleVectorStore:
    """最简单的向量存储 + 检索实现"""
    
    def __init__(self, embedding_model: str = "text-embedding-3-small"):
        self.client = OpenAI()  # 使用 OpenAI Embedding API
        self.embedding_model = embedding_model
        self.chunks = []       # 原始文本块
        self.embeddings = []   # 对应的向量
    
    def embed(self, texts: List[str]) -> np.ndarray:
        """将文本转为向量"""
        response = self.client.embeddings.create(
            model=self.embedding_model,
            input=texts
        )
        return np.array([d.embedding for d in response.data])
    
    def add_documents(self, chunks: List[str]):
        """添加文档到向量存储"""
        print(f"正在向量化 {len(chunks)} 个文档块...")
        self.chunks.extend(chunks)
        embeddings = self.embed(chunks)
        self.embeddings.extend(embeddings)
        self.embeddings = np.array(self.embeddings)
        print(f"完成！向量维度: {self.embeddings.shape}")
    
    def search(self, query: str, top_k: int = 5) -> List[tuple]:
        """语义搜索：返回最相关的 top_k 个文档块"""
        query_vec = self.embed([query])[0]
        
        # 计算余弦相似度
        similarities = np.dot(self.embeddings, query_vec) / (
            np.linalg.norm(self.embeddings, axis=1) * np.linalg.norm(query_vec)
        )
        
        # 获取 top_k
        top_indices = np.argsort(similarities)[-top_k:][::-1]
        
        results = []
        for idx in top_indices:
            results.append((self.chunks[idx], float(similarities[idx])))
        return results

# 使用示例
store = SimpleVectorStore()
store.add_documents(chunks)

results = store.search("什么是 Transformer 的自注意力机制？", top_k=3)
for i, (chunk, score) in enumerate(results):
    print(f"\n--- 结果 {i+1} (相似度: {score:.3f}) ---")
    print(chunk[:300] + "...")

Step 3：组装 RAG 管线

# rag.py —— 完整的 RAG 问答管线
class RAGPipeline:
    """检索增强生成管线"""
    
    def __init__(self, vector_store: SimpleVectorStore, 
                 llm_model: str = "gpt-4o-mini"):
        self.store = vector_store
        self.llm_model = llm_model
        self.llm_client = OpenAI()
    
    def build_prompt(self, query: str, contexts: List[str]) -> str:
        """构建包含上下文的提示词"""
        context_text = "\n\n---\n\n".join(
            f"[来源 {i+1}]\n{ctx}" 
            for i, ctx in enumerate(contexts)
        )
        
        return f"""你是一个知识渊博的 AI 助手。请基于以下参考资料回答用户问题。

## 参考资料
{context_text}

## 回答规则
1. 如果参考资料中有答案，请基于参考资料回答，并标注来源编号
2. 如果参考资料不足以回答，请诚实说明
3. 不要编造参考资料中没有的信息

## 用户问题
{query}

## 你的回答""" 
    
    def query(self, question: str, top_k: int = 3) -> dict:
        """执行 RAG 查询"""
        # 1. 检索相关文档
        search_results = self.store.search(question, top_k=top_k)
        contexts = [chunk for chunk, score in search_results if score > 0.3]
        scores = [score for _, score in search_results if score > 0.3]
        
        if not contexts:
            return {
                "answer": "抱歉，没有找到与您问题相关的参考资料。",
                "sources": [],
                "scores": []
            }
        
        # 2. 构建提示词
        prompt = self.build_prompt(question, contexts)
        
        # 3. 调用 LLM 生成回答
        response = self.llm_client.chat.completions.create(
            model=self.llm_model,
            messages=[{"role": "user", "content": prompt}],
            temperature=0.3
        )
        
        return {
            "answer": response.choices[0].message.content,
            "sources": contexts,
            "scores": scores
        }

# 使用示例
rag = RAGPipeline(store)
result = rag.query("什么是 Transformer？", top_k=3)
print("回答:", result["answer"])
print("\n使用的来源数量:", len(result["sources"]))
🔑 关键要点：
1. RAG 的核心就是三步：分块 → 向量化 → 检索 → 组装 → 生成，没有魔法
2. 分块策略直接影响检索质量——太大检索不准，太小丢失上下文
3. chunk_overlap 是关键参数：防止关键信息恰好在两个块的边界被切断
4. 这个 200 行的实现已经可以处理真实的文档问答，LangChain 只是把这些步骤封装了起来
5. 生产环境还需要：重排序（rerank）、混合检索（关键词+语义）、缓存策略
