固件代码 — 轮式底盘 (Arduino)
cell_wheel.ino C++ 
// 细胞 Cell — 轮式底盘固件 v1.0
// 接收手机端指令驱动电机 + 超声波避障
// 硬件: Arduino Nano + L298N + HC-SR04

#define ENA 5   // 右电机PWM
#define IN1 6   // 右电机正转
#define IN2 7   // 右电机反转
#define ENB 9   // 左电机PWM
#define IN3 10  // 左电机正转
#define IN4 11  // 左电机反转
#define TRIG 12  // 超声波触发
#define ECHO 13  // 超声波回波

int speed = 200;  // 默认速度 0-255

void setup() {
  Serial.begin(115200);
  pinMode(ENA, OUTPUT); pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT); pinMode(ENB, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  pinMode(TRIG, OUTPUT); pinMode(ECHO, INPUT);
}

void loop() {
  // 读取超声波距离
  long dist = readSonar();
  Serial.print("SENSOR:dist:");
  Serial.println(dist);

  // 自动避障: 前方<20cm停止
  if (dist > 0 && dist < 20) {
    stopMotor();
    Serial.println("CMD:obstacle!");
  }

  // 解析手机指令
  if (Serial.available()) {
    String cmd = Serial.readStringUntil('\n');
    parseCommand(cmd);
  }
  delay(50);
}

long readSonar() {
  digitalWrite(TRIG, LOW); delayMicroseconds(2);
  digitalWrite(TRIG, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  long dur = pulseIn(ECHO, HIGH, 30000);
  return dur / 58;  // 换算cm
}

void parseCommand(String cmd) {
  // 指令格式: CMD:forward / CMD:back / CMD:left
  //          CMD:right / CMD:stop / CMD:speed:200
  if (cmd == "CMD:forward") forward();
  else if (cmd == "CMD:back") backward();
  else if (cmd == "CMD:left") turnLeft();
  else if (cmd == "CMD:right") turnRight();
  else if (cmd == "CMD:stop") stopMotor();
  else if (cmd.startsWith("CMD:speed:")) {
    speed = cmd.substring(10).toInt();
  }
}

void forward() {
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
  analogWrite(ENA, speed); analogWrite(ENB, speed);
}

void stopMotor() {
  analogWrite(ENA, 0); analogWrite(ENB, 0);
}
固件代码 — 机械臂 (ESP32 BLE)
cell_arm.ino C++ 
// 细胞 Cell — 机械臂固件 v1.0
// BLE接收手机指令驱动4个SG90舵机
// 硬件: ESP32 + SG90×4

#include <ESP32Servo.h>
#include <BLEDevice.h>
#include <BLEServer.h>

Servo base, arm1, arm2, gripper;
int pos[4] = {90, 90, 90, 0};

#define SERVICE_UUID "4fafc201-1fb5-459e"
#define CHAR_UUID    "beb5483e-36e1-4688"

void setup() {
  Serial.begin(115200);
  base.attach(25, 500, 2500);
  arm1.attach(26, 500, 2500);
  arm2.attach(27, 500, 2500);
  gripper.attach(14, 500, 2500);
  updateServos();

  // 初始化BLE
  BLEDevice::init("Cell-Arm-02");
  BLEServer* server = BLEDevice::createServer();
  BLEService* svc = server->createService(SERVICE_UUID);
  BLECharacteristic* chr = svc->createCharacteristic(
    CHAR_UUID, BLECharacteristic::PROPERTY_WRITE);
  svc->start();
  BLEDevice::startAdvertising();
}

void loop() {
  // BLE指令处理在回调中
  delay(20);
}

// 处理手机发来的指令
// 格式: "ARM:base:90" / "ARM:grip:1" / "ARM:grab"
void onCommand(String cmd) {
  if (cmd == "ARM:grab") {
    // 自动抓取序列
    smoothMove(gripper, pos[3], 45);
    pos[3] = 45;
  } else if (cmd == "ARM:release") {
    smoothMove(gripper, pos[3], 0);
    pos[3] = 0;
  } else if (cmd.startsWith("ARM:")) {
    // 解析关节指令 ARM:joint:angle
    int joint = cmd.substring(4, 5).toInt();
    int angle = cmd.substring(6).toInt();
    pos[joint] = constrain(angle, 0, 180);
    updateServos();
  }
}

void smoothMove(Servo& s, int from, int to) {
  int step = from < to ? 1 : -1;
  for (int i = from; i != to; i += step) {
    s.write(i); delay(15);
  }
}

void updateServos() {
  base.write(pos[0]); arm1.write(pos[1]);
  arm2.write(pos[2]); gripper.write(pos[3]);
}

→ 手机→Arduino (USB串口):
