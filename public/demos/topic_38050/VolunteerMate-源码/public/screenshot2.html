好的，我来设计一个 打分制推荐算法。核心思路：

分类偏好得分：统计用户历史打卡中各分类（环保/捐赠/帮扶/传播）的占比，同类任务加分
时间段匹配：早/晚时间短任务（15min内）权重高；下午优先推荐深度任务（45min+）
难度-等级匹配：新手用户推荐简单任务；高等级用户优先推荐挑战性任务
新任务探索奖励：从未做过的类型额外加分，引导拓展
近期重复惩罚：近 3 天内做过的任务扣分，避免重复
今日主题加成：与今日主题同类的任务额外加分（权重最高）

📊 算法架构：

用户数据(UserStats) 今日主题 当前时间段 │ │ │ ▼ ▼ ▼ ┌─────────────────────────────────────────────────┐ │ 综 合 打 分 函 数 │ │ 分类偏好 + 时间段 + 难度匹配 + 主题加成 │ │ + 新任务奖励 - 近期重复惩罚 │ └────────────────────┬────────────────────────────────┘ │ 排序 → Top 3 任务 → 生成个性化理由 │ ▼ 最终推荐：[{ task, score, reason }]

💡 去重策略：按得分排序后，检查 Top 3 中分类重复度。若有同类型，用"跳过同类已选"的方式插入下一个得分最高的异类任务，确保推荐的多样性。

// 公益任务推荐引擎 import { Task, ALL_TASKS } from '../data/tasks'; import { UserStats } from '../store/useStore'; interface Recommendation { task: Task; score: number; reason: string; } // 时间段判断 function getTimeOfDay(): 'morning' | 'afternoon' | 'evening' { const h = new Date().getHours(); return h < 11 ? 'morning' : h < 18 ? 'afternoon' : 'evening'; } // 分类偏好得分：用户做得越多的类型，同类型任务加分越多 function getCategoryPreference(stats: UserStats): Record<string, number> { const total = stats.totalCheckIns || 1; return { '环保': (stats.environmentalTasks || 0) / total, '捐赠': (stats.donationTasks || 0) / total, '帮扶': (stats.helpTasks || 0) / total, '传播': (stats.spreadTasks || 0) / total, }; } export function getAITaskRecommendations( stats: UserStats, recentTaskIds: string[] = [], theme: string = '环保', count: number = 3 ): Recommendation[] { const timeOfDay = getTimeOfDay(); const categoryPref = getCategoryPreference(stats); // 对每个任务计算推荐得分 const scored = ALL_TASKS.map(task => { let score = 0; const reasons: string[] = []; // 1) 今日主题加成（权重最高 +20） if (task.category === theme) { score += 20; reasons.push(`${theme}是今日主题`); } // 2) 分类偏好（+0~15 分） const pref = categoryPref[task.category] || 0; if (pref > 0.3) { score += 15; reasons.push(`你${task.category}类任务参与度最高`); } // 3) 时间段匹配 if ((timeOfDay === 'morning' || timeOfDay === 'evening') && task.duration <= 20) { score += 10; reasons.push('这个时间段适合快速完成'); } else if (timeOfDay === 'afternoon' && task.duration >= 30) { score += 12; reasons.push('下午有充足时间深入参与'); } // 4) 新任务探索奖励（从未做过的+8） if (pref === 0) { score += 8; reasons.push('探索新类型，拓宽公益体验'); } // 5) 近期重复惩罚（近3天做过 -25） if (recentTaskIds.includes(task.id)) { score -= 25; } // 6) 难度-等级匹配 const level = getLevel(stats.totalMinutes); if (level >= 4 && task.difficulty === '挑战') { score += 10; reasons.push('你的等级已足够挑战'); } if (level <= 2 && task.difficulty === '简单') { score += 8; reasons.push('新手友好，轻松入门'); } return { task, score, reason: reasons.slice(0, 2).join('；') }; }); // 去重排序（确保Top3中每类最多1个任务） scored.sort((a, b) => b.score - a.score); const result: Recommendation[] = []; const used = new Set<string>(); for (const item of scored) { if (used.has(item.task.category) && result.length < count - 1) continue; result.push(item); used.add(item.task.category); if (result.length >= count) break; } return result; }
